 14def logBeta(alpha):
 15    '''Returns the logarithm of the multivariate beta function.
 16
 17    **Arguments:**
 18
 19    * **alpha** (`list`[`float`]): A list of arguments for the function.
 20
 21    **Returns:**
 22
 23    * `float`: The the natural logarithm of value of the multivariate
 24        beta function, evaluated at the given list of arguments.
 25
 26    '''
 27    return np.sum(gammaln(alpha)) - gammaln(np.sum(alpha))

 33class gamer_gen(rv_continuous):
 34    def _pdf(self, x, r, a):
 35        return r / a ** r * np.exp(gammaln(a + r) - gammaln(a)) * (
 36            x ** (-r - 1) * gammainc(a + r, a * x)
 37        )
 38
 39    def _rvs(self, r, a, size=None, random_state=None):
 40        meanScores = pareto.rvs(
 41            r,
 42            size=size,
 43            random_state=random_state
 44        )
 45        return gamma.rvs(
 46            a,
 47            scale=meanScores / a,
 48            random_state=random_state
 49        )
 50
 51    def _munp(self, n, r, a):
 52        if r <= n:
 53            return np.inf
 54        return r / a ** n * np.exp(gammaln(a + n) - gammaln(a)) / (r - n)

 60gamer = gamer_gen(a=0.0, name='gamer')
 61r'''A gamer distribution as in Section 6.5 of the Paper.
 62
 63As an instance of the `scipy.stats.rv_continuous` class, `gamer` object
 64inherits from it a collection of generic methods (see below for the full
 65list), and completes them with details specific for this particular
 66distribution.
 67
 68The probability density function of `gamer` is:
 69$$
 70  f(x) = \frac r {a^r \, \Gamma(a)} \,
 71    x^{-r - 1} \, \int_0^{ax} y^{a + r - 1} e^{-y} \, dy
 72$$
 73for $x \ge 0$, $r > 0$, and $a > 0$. $\Gamma$ is the gamma function
 74(`scipy.special.gamma`).
 75
 76`gamer` takes $r$ and $a$ as shape parameters.
 77
 78The probability density above is defined in the "standardized" form. To
 79shift and/or scale the distribution use the `loc` and `scale`
 80parameters. Specifically, `gamer.pdf(x, r, a, loc, scale)` is
 81identically equivalent to `gamer.pdf(y, r, a) / scale` with
 82`y = (x - loc) / scale`.
 83
 84In the notation of the Paper, `gamer` represents the
 85$\mathrm{Gamer}(r, 1, a)$ distribution. To obtain the
 86$\mathrm{Gamer}(r, c, a)$ distribution, set the `scale` parameter to
 87$c$.
 88
 89**Methods:**
 90
 91* **rvs(r, a, loc=0, scale=1, size=1, random_state=None)**:
 92    Random variates.
 93* **pdf(x, r, a, loc=0, scale=1)**:
 94    Probability density function.
 95* **logpdf(x, r, a, loc=0, scale=1)**:
 96    Log of the probability density function.
 97* **cdf(x, r, a, loc=0, scale=1)**:
 98    Cumulative distribution function.
 99* **logcdf(x, r, a, loc=0, scale=1)**:
100    Log of the cumulative distribution function.
101* **sf(x, r, a, loc=0, scale=1)**:
102    Survival function (also defined as ``1 - cdf``).
103* **logsf(x, r, a, loc=0, scale=1)**:
104    Log of the survival function.
105* **ppf(q, r, a, loc=0, scale=1)**:
106    Percent point function (inverse of ``cdf`` --- percentiles).
107* **isf(q, r, a, loc=0, scale=1)**:
108    Inverse survival function (inverse of ``sf``).
109* **moment(order, r, a, loc=0, scale=1)**:
110    Non-central moment of the specified order.
111* **stats(r, a, loc=0, scale=1, moments='mv')**:
112    Mean('m'), variance('v'), skew('s'), and/or kurtosis('k').
113* **entropy(r, a, loc=0, scale=1)**:
114    (Differential) entropy of the RV.
115* **fit(data)**:
116    Parameter estimates for generic data. See
117    `scipy.stats.rv_continuous.fit` for detailed documentation of the
118    keyword arguments.
119* **expect(func, args=(r, a), loc=0, scale=1, lb=None, ub=None,
120conditional=False, **kwds)**:
121    Expected value of a function (of one argument) with respect to the
122    distribution.
123* **median(r, a, loc=0, scale=1)**:
124    Median of the distribution.
125* **mean(r, a, loc=0, scale=1)**:
126    Mean of the distribution.
127* **var(r, a, loc=0, scale=1)**:
128    Variance of the distribution.
129* **std(r, a, loc=0, scale=1)**:
130    Standard deviation of the distribution.
131* **interval(confidence, r, a, loc=0, scale=1)**:
132    Confidence interval with equal areas around the median.

136class Measure:
137    r'''A finite, signed Borel measure on the real line.
138
139    `Measure` objects can be added to each other and multiplied on the
140    left by a `float`. These operations return new `Measure` objects.
141    When adding two instances, the `bandwidth` of the result is the
142    maximum of the bandwidths of the summands.
143
144    '''
145    def __init__(self, cdf, pdf, mean=None, totalMass=1):
146        '''Constructs a `Measure` object from the given arguments.
147
148        '''
149        self._cdf = cdf
150        self._pdf = pdf
151        self._mean = mean
152        self._totalMass = totalMass
153
154    @property
155    def cdf(self):
156        r'''`func`: The distribution function of the measure. If $\mu$
157        is the measure, then this is the function $t \mapsto
158        \mu((-\infty, t])$.
159        '''
160        return self._cdf
161
162    @property
163    def pdf(self):
164        '''`func`: The density of the measure, which is the derivative
165        of `cdf`. If `cdf` is not differentiable, then an approximate
166        density should be provided.
167        '''
168        return self._pdf
169
170    @property
171    def mean(self):
172        r'''`float` or `None`: If the mean is stored in the `Measure`
173        instance, then it should equal $\int_{-\infty}^\infty x \,
174        dF(x)$, where $F$ is the distribution function. The mean of the
175        sum of two `Measure` instances will be `None` if either mean is
176        `None`.
177        '''
178        return self._mean
179
180    @property
181    def totalMass(self):
182        r'''`float`: The total mass of the measure. If $\mu$ is the
183        measure, then this $\mu(\mathbb{R})$.
184        '''
185        return self._totalMass
186
187    @property
188    def bandwidth(self):
189        '''`float`: If `pdf` is an approximate density created with the
190        `gaussian_kde` function from the `scipy.stats` module, then this
191        is the bandwidth that was used to create it. Is 0 otherwise. See
192        `NDPModel.law` for more details.
193        '''
194        try:
195            return self.pdf.factor
196        except AttributeError:
197            return 0
198
199    def __rmul__(self, scalar):
200        def pdf(t):
201            return scalar * self.pdf(t)
202        pdf.factor = self.bandwidth
203        mean = (
204            None if self._mean is None
205            else scalar * self._mean
206        )
207        return Measure(
208            lambda t: scalar * self._cdf(t),
209            pdf,
210            mean,
211            scalar * self._totalMass
212        )
213
214    def __add__(self, other):
215        def pdf(t):
216            return self._pdf(t) + other.pdf(t)
217        pdf.factor = max(self.bandwidth, other.bandwidth)
218        mean = (
219            None if self._mean is None or other.mean is None
220            else self._mean + other.mean
221        )
222        return Measure(
223            lambda t: self._cdf(t) + other.cdf(t),
224            pdf,
225            mean,
226            self._totalMass + other.totalMass
227        )

229class NDPModel:
230    r'''An NDP model on the state space $S = \\{0, ..., L - 1\\}$.
231
232    The NDP model consists of both an NDP and some observed data. See
233    Section 6.1 of the Paper for details.
234
235    '''
236
237    def __init__(self, colConc, rowConc, baseMeas, rowCounts):
238        '''Uses the given arguments to set the corresponding properties.
239        The `baseMeas` and `rowCounts` arguments should be numpy arrays
240        of the same shape as the corresponding properties.
241
242        '''
243        self._ndp = {
244            'colConc': colConc,
245            'rowConc': rowConc,
246            'baseMeas': baseMeas
247        }
248        self._rowCounts = rowCounts
249
250        rowConc = self._ndp['rowConc']
251        baseMeas = self._ndp['baseMeas']
252        logDenom = logBeta(rowConc * baseMeas)
253        self._likelihoods = [
254            np.exp(logBeta(rowConc * baseMeas + row) - logDenom)
255            for row in rowCounts
256        ]
257
258        self._logScale = 0
259        self._weightedSims = []
260
261        self._savedSims = []
262        '''The state of `self._weightedSims` just prior to the last
263        successful call to `setEss`. Is reset by `addSims`, `clearSims`,
264        and `restoreSims`.
265        '''
266
267        self._allSims = []
268        '''The list of all simulations ever created by `addSims`.'''

278    def rowConc(self):

290        '''

327        r'''`tuple`(`WeightedSim`): A tuple of distinct (and therefore
328        independent) `WeightedSim` instances. `weightedSims`[k] is
329        denoted by $(t^k, \boldsymbol{\theta}_M^{*, k})$ in Section 6.1

335    @property
336    def ess(self):
337        '''`float`: The effective sample size, approximated with the
338        sample variance. Denoted by $K_e''$ in Section 4.2 of the Paper.
339        '''
340        numSims = len(self._weightedSims)
341        if numSims == 0:
342            return 0
343        essPop = (
344            sum(sim.totalWeight      for sim in self._weightedSims) ** 2 /
345            sum(sim.totalWeight ** 2 for sim in self._weightedSims)
346        )
347        if essPop == numSims:
348            return numSims
349        return essPop * (numSims - 1) / (numSims - essPop / numSims)

382    def setEss(self, minEss, maxDel=None):
383        '''Tries to set the `ess` to the given value by deleting
384        simulations from `weightedSims`. Simulations are deleted in
385        order of their `WeightedSim.totalWeight` property, starting with
386        the largest.
387
388        **Arguments:**
389
390        * **minEss** (`float`): The minimum value of `ess` to achieve.
391            The method will stop deleting simulations when `ess` reaches
392            or exceeds this value.
393        * **maxDel** (`int`): If provided, the method will not delete
394            more than this number of simulations.
395
396        **Returns:**
397
398        * `int`: The number of simulations deleted.
399
400        **Raises:**
401
402        * **ValueError:** If
403            1. `minEss` does not exceed `ess`;
404            2. `minEss` is not less than the total number of simulations
405                in `weightedSims`;
406            3. `maxDel` is negative or zero; or
407            4. the method is unable to achieve the given value for
408                `ess`. In this case, all deletions are reversed and the
409                `weightedSims` property is left unaltered.
410
411        '''
412        savedSims = self._weightedSims.copy()
413        # Temporarily stored. Will be transferred to `self._savedSims`
414        # if the method finished without error.
415
416        if self.ess >= minEss:
417            raise ValueError(f'ESS is already {self.ess}')
418        if minEss >= len(self._weightedSims):
419            raise ValueError('Not enough simulations')
420        if maxDel is not None and maxDel <= 0:
421            raise ValueError('`maxDel` must be positive')
422
423        # make `minKeep`, the minimum number of simulations to keep
424        if maxDel is None:
425            minKeep = minEss
426        else:
427            minKeep = len(self._weightedSims) - maxDel
428
429        # try to reach `minEss` without deleting too many sims
430        sortedSims = sorted(
431            self._weightedSims,
432            key=lambda sim: sim.totalWeight,
433            reverse=True
434        )
435        for sim in sortedSims:
436            self._weightedSims.remove(sim)
437            if self.ess >= minEss or len(self._weightedSims) <= minKeep:
438                break
439
440        # check for success
441        if self.ess < minEss:
442            self._weightedSims = savedSims # undo deletions
443            raise ValueError(f'Cannot achieve an ESS of {minEss}')
444
445        self._savedSims = savedSims # on success, save for undo
446
447        return len(savedSims) - len(self._weightedSims)

465    def prior(self, mapping, numSamples=10000, bandwidth=None):
466        r'''Let the random vector $(X_0, X_1, \ldots, X_{L - 1})$ have a
467        Dirichlet distribution with parameters `colConc` * `baseMeas`.
468        This method returns the distribution of a random variable of the
469        form $Y = f(X_0, X_1, \ldots, X_{L - 1})$. The distribution of
470        $Y$ is a Borel probability measure on the real line. This method
471        therefore returns a `Measure` object.
472
473        If a function is passed to the `mapping` argument, then the cdf
474        and pdf of $Y$ are estimated by sampling and Gaussian kernel
475        density estimation (KDE). See `agentLaw` for details about
476        KDE.
477
478        This method is used in the implementation of (8.24), (8.26),
479        (9.6), and the final displayed equation in Section 9 of the
480        paper.
481
482        **Arguments:**
483
484        * **mapping** (`func` or `int`): The function to apply to the
485            random vector $(X_0, X_1, \ldots, X_{L - 1})$. Denoted by
486            $f$ in the above description. Passing this argument an
487            integer $\ell$ is equivalent to using the function $f(X_0,
488            X_1, \ldots, X_{L - 1}) = X_\ell$.
489        * **numSamples** (`int`): The number of samples to use when
490            estimating the cdf and pdf.
491        * **bandwidth** (`float`): The bandwidth to use in the KDE.
492
493        **Returns:**
494
495        * `Measure`: The distribution of $Y$.
496
497        '''
498        if not callable(mapping):
499            prior = beta(
500                self._ndp['rowConc'] * self._ndp['baseMeas'][mapping],
501                self._ndp['rowConc'] * (1 - self._ndp['baseMeas'][mapping])
502            )
503            return Measure(prior.cdf, prior.pdf, prior.mean())
504        rawSamples = dirichlet.rvs(
505            self._ndp['rowConc'] * self._ndp['baseMeas'],
506            numSamples
507        )
508        samples = [mapping(sample) for sample in rawSamples]
509        pdf = gaussian_kde(samples, bandwidth)
510        return Measure(
511            ecdf(samples).cdf.evaluate,
512            pdf,
513            np.average(samples),
514            pdf.factor
515        )

517    def law(self, mapping, newAgent=False, bandwidth=None):
518        r'''Let $\theta_{m\ell}$ be the unknown probability that agent
519        $m$ will perform action $\ell$, and let $\theta$ be the matrix
520        with entries $\theta_{m\ell}$. Let $\vartheta_\ell$ be the
521        unknown probability that a new, unobserved agent will perform
522        action $\ell$ This method returns the law (or distribution) of a
523        random variable of the form $\Theta = f(\vartheta_0, \ldots,
524        \vartheta_{L - 1})$, or of the form $\Theta = f(\theta)$. To get
525        the former, set `newAgent` to `True`. To get the latter, set
526        `newAgent` to `False`. In either case, set `mapping` to $f$.
527
528        The law of $\Theta$ is a Borel probability measure on the
529        real line. This method therefore returns a `Measure` object.
530        However, the law of $\Theta$ has a discrete component, so
531        it does not have a density. The `Measure.pdf` attribute of the
532        returned object is an approximate density, created with Gaussian
533        kernel density estimation (KDE). More specifically, it treats a
534        point mass $\delta_x$ as having a Gaussian density with mean $x$
535        and standard deviation $h$, where $h$ is the "bandwidth" of the
536        estimate.
537
538        This method implements either (5.12) or (5.10) in the Paper,
539        depending on whether `newAgent` is `True` or `False`,
540        respectively.
541
542        **Arguments:**
543
544        * **mapping** (`func`): The function $f$ in the above
545            description. This function should take a single `ndarray`
546            object as an argument. If `newAgent` is `True`, the array
547            should have the same shape as `baseMeas`. Otherwise, it
548            should have the same shape as `rowCounts`.
549        * **newAgent** (`bool`): Determines the form of $\Theta$ in the
550            above description.
551        * **bandwidth** (`float` or `None`): The bandwidth to use in the
552            Gaussian KDE. If `None`, the bandwidth will be calculated
553            using [Scott's Rule](https://doi.org/10.1002/9780470316849).
554            The calculated value can be retrieved by accessing the
555            `Measure.bandwidth` property of the returned object. This
556            argument is passed directly to the `gaussian_kde` function
557            in the `scipy.stats` module. See that documentation for
558            further details.
559
560        **Returns:**
561
562        * `Measure`: The law of $\Theta$.
563
564        **Raises:**
565
566        * **ValueError**: If the `weightedSims` property is empty.
567
568        '''
569        if not self._weightedSims:
570            raise ValueError('No weighted simulations found')
571
572        numRows = len(self._rowCounts)
573
574        if not newAgent:
575            # Implement (8.10).
576            samples = np.array([
577                mapping(weightedSim.rowDistSims)
578                for weightedSim in self._weightedSims
579            ])
580            weights = np.array([
581                weightedSim.totalWeight
582                for weightedSim in self._weightedSims
583            ])
584        else:
585            # Implement (8.24). In this case, the `sample` list is
586            # ordered as
587            # \[
588            #   \theta_1^{*, 1}, \ldots, \theta_M^{*, 1},
589            #   \theta_1^{*, 2}, \ldots, \theta_M^{*, 2},
590            #   \ldots,
591            #   \theta_1^{*, K}, \ldots, \theta_M^{*, K}
592            # \]
593            samples = np.array([
594                mapping(weightedSim.rowDistSims[m])
595                for weightedSim in self._weightedSims # outer loop
596                for m in range(numRows)  # inner loop
597            ])
598            weights = np.concatenate([
599                [weightedSim.totalWeight] * numRows
600                for weightedSim in self._weightedSims
601            ])
602
603        # Create the sample law
604        sampleCDF = ECDFDiscrete(samples, weights)
605        samplePDF = gaussian_kde(samples, bandwidth, weights)
606        sampleLaw = Measure(
607            sampleCDF,
608            samplePDF,
609            np.average(samples, weights=weights),
610            samplePDF.factor
611        )
612
613        # If implementing (8.10), return only the sample law. Otherwise,
614        # return a mixture as in (8.24).
615        if not newAgent:
616            return sampleLaw
617        colConc = self._ndp['colConc']
618        return (
619            (colConc / (colConc + numRows)) * self.prior(mapping) +
620            (numRows / (colConc + numRows)) * sampleLaw
621        )

As an instance of the scipy.stats.rv_continuous class, gamer object inherits from it a collection of generic methods (see below for the full list), and completes them with details specific for this particular distribution.

The probability density function of gamer is: $$ f(x) = \frac r {a^r \, \Gamma(a)} \, x^{-r - 1} \, \int_0^{ax} y^{a + r - 1} e^{-y} \, dy $$ for $x \ge 0$, $r > 0$, and $a > 0$. $\Gamma$ is the gamma function (scipy.special.gamma).

689        self._rowWeights = []
690        self._rowDistSims = []

692

701

710                else:
