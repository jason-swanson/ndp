 14def logBeta(alpha):
 15    '''Returns the logarithm of the multivariate beta function.
 16
 17    **Arguments:**
 18
 19    * **alpha** (`list`[`float`]): A list of arguments for the function.
 20
 21    **Returns:**
 22
 23    * `float`: The the natural logarithm of value of the multivariate
 24        beta function, evaluated at the given list of arguments.
 25
 26    '''
 27    return np.sum(gammaln(alpha)) - gammaln(np.sum(alpha))

 33class gamer_gen(rv_continuous):
 34    def _pdf(self, x, r, a):
 35        return r / a ** r * np.exp(gammaln(a + r) - gammaln(a)) * (
 36            x ** (-r - 1) * gammainc(a + r, a * x)
 37        )
 38
 39    def _rvs(self, r, a, size=None, random_state=None):
 40        meanScores = pareto.rvs(
 41            r,
 42            size=size,
 43            random_state=random_state
 44        )
 45        return gamma.rvs(
 46            a,
 47            scale=meanScores / a,
 48            random_state=random_state
 49        )
 50
 51    def _munp(self, n, r, a):
 52        if r <= n:
 53            return np.inf
 54        return r / a ** n * np.exp(gammaln(a + n) - gammaln(a)) / (r - n)

 60gamer = gamer_gen(a=0.0, name='gamer')
 61r'''A gamer distribution as in Section 9 of the Paper.
 62
 63As an instance of the `scipy.stats.rv_continuous` class, `gamer` object
 64inherits from it a collection of generic methods (see below for the full
 65list), and completes them with details specific for this particular
 66distribution.
 67
 68The probability density function of `gamer` is:
 69$$
 70  f(x) = \frac r {a^r \, \Gamma(a)} \,
 71    x^{-r - 1} \, \int_0^{ax} y^{a + r - 1} e^{-y} \, dy
 72$$
 73for $x \ge 0$, $r > 0$, and $a > 0$. $\Gamma$ is the gamma function
 74(`scipy.special.gamma`).
 75
 76`gamer` takes $r$ and $a$ as shape parameters.
 77
 78The probability density above is defined in the "standardized" form. To
 79shift and/or scale the distribution use the `loc` and `scale`
 80parameters. Specifically, `gamer.pdf(x, r, a, loc, scale)` is
 81identically equivalent to `gamer.pdf(y, r, a) / scale` with
 82`y = (x - loc) / scale`.
 83
 84In the notation of the Paper, `gamer` represents the
 85$\mathrm{Gamer}(r, 1, a)$ distribution. To obtain the
 86$\mathrm{Gamer}(r, c, a)$ distribution, set the `scale` parameter to
 87$c$.
 88
 89**Methods:**
 90
 91* **rvs(r, a, loc=0, scale=1, size=1, random_state=None)**:
 92    Random variates.
 93* **pdf(x, r, a, loc=0, scale=1)**:
 94    Probability density function.
 95* **logpdf(x, r, a, loc=0, scale=1)**:
 96    Log of the probability density function.
 97* **cdf(x, r, a, loc=0, scale=1)**:
 98    Cumulative distribution function.
 99* **logcdf(x, r, a, loc=0, scale=1)**:
100    Log of the cumulative distribution function.
101* **sf(x, r, a, loc=0, scale=1)**:
102    Survival function (also defined as ``1 - cdf``).
103* **logsf(x, r, a, loc=0, scale=1)**:
104    Log of the survival function.
105* **ppf(q, r, a, loc=0, scale=1)**:
106    Percent point function (inverse of ``cdf`` --- percentiles).
107* **isf(q, r, a, loc=0, scale=1)**:
108    Inverse survival function (inverse of ``sf``).
109* **moment(order, r, a, loc=0, scale=1)**:
110    Non-central moment of the specified order.
111* **stats(r, a, loc=0, scale=1, moments='mv')**:
112    Mean('m'), variance('v'), skew('s'), and/or kurtosis('k').
113* **entropy(r, a, loc=0, scale=1)**:
114    (Differential) entropy of the RV.
115* **fit(data)**:
116    Parameter estimates for generic data. See
117    `scipy.stats.rv_continuous.fit` for detailed documentation of the
118    keyword arguments.
119* **expect(func, args=(r, a), loc=0, scale=1, lb=None, ub=None,
120conditional=False, **kwds)**:
121    Expected value of a function (of one argument) with respect to the
122    distribution.
123* **median(r, a, loc=0, scale=1)**:
124    Median of the distribution.
125* **mean(r, a, loc=0, scale=1)**:
126    Mean of the distribution.
127* **var(r, a, loc=0, scale=1)**:
128    Variance of the distribution.
129* **std(r, a, loc=0, scale=1)**:
130    Standard deviation of the distribution.
131* **interval(confidence, r, a, loc=0, scale=1)**:
132    Confidence interval with equal areas around the median.

136class Measure:
137    r'''A finite, signed Borel measure on the real line.
138
139    `Measure` objects can be added to each other and multiplied on the
140    left by a `float`. These operations return new `Measure` objects.
141    When adding two instances, the `bandwidth` of the result is the
142    maximum of the bandwidths of the summands.
143
144    '''
145    def __init__(self, cdf, pdf, mean=None, totalMass=1):
146        '''Constructs a `Measure` object from the given arguments.
147
148        '''
149        self._cdf = cdf
150        self._pdf = pdf
151        self._mean = mean
152        self._totalMass = totalMass
153
154    @property
155    def cdf(self):
156        r'''`func`: The distribution function of the measure. If $\mu$
157        is the measure, then this is the function $t \mapsto
158        \mu((-\infty, t])$.
159        '''
160        return self._cdf
161
162    @property
163    def pdf(self):
164        '''`func`: The density of the measure, which is the derivative
165        of `cdf`. If `cdf` is not differentiable, then an approximate
166        density should be provided.
167        '''
168        return self._pdf
169
170    @property
171    def mean(self):
172        r'''`float` or `None`: If the mean is stored in the `Measure`
173        instance, then it should equal $\int_{-\infty}^\infty x \,
174        dF(x)$, where $F$ is the distribution function. The mean of the
175        sum of two `Measure` instances will be `None` if either mean is
176        `None`.
177        '''
178        return self._mean
179
180    @property
181    def totalMass(self):
182        r'''`float`: The total mass of the measure. If $\mu$ is the
183        measure, then this $\mu(\mathbb{R})$.
184        '''
185        return self._totalMass
186
187    @property
188    def bandwidth(self):
189        '''`float`: If `pdf` is an approximate density created with the
190        `gaussian_kde` function from the `scipy.stats` module, then this
191        is the bandwidth that was used to create it. Is 0 otherwise. See
192        `IDPModel.law` for more details.
193        '''
194        try:
195            return self.pdf.factor
196        except AttributeError:
197            return 0
198
199    def __rmul__(self, scalar):
200        def pdf(t):
201            return scalar * self.pdf(t)
202        pdf.factor = self.bandwidth
203        mean = (
204            None if self._mean is None
205            else scalar * self._mean
206        )
207        return Measure(
208            lambda t: scalar * self._cdf(t),
209            pdf,
210            mean,
211            scalar * self._totalMass
212        )
213
214    def __add__(self, other):
215        def pdf(t):
216            return self._pdf(t) + other.pdf(t)
217        pdf.factor = max(self.bandwidth, other.bandwidth)
218        mean = (
219            None if self._mean is None or other.mean is None
220            else self._mean + other.mean
221        )
222        return Measure(
223            lambda t: self._cdf(t) + other.cdf(t),
224            pdf,
225            mean,
226            self._totalMass + other.totalMass
227        )

229class IDPModel:
230    r'''An IDP model on the state space $S = \\{0, ..., L - 1\\}$.
231
232    The IDP model consists of both an IDP and some observed data. See
233    Section 9 of the Paper for details. The bijection $\sigma$ that
234    appears in the Paper is taken here to be the identity.
235
236    '''
237
238    def __init__(self, colConc, rowConc, baseMeas, rowCounts):
239        '''Uses the given arguments to set the corresponding properties.
240        The `baseMeas` and `rowCounts` arguments should be numpy arrays
241        of the same shape as the corresponding properties.
242
243        '''
244        self._idp = {
245            'colConc': colConc,
246            'rowConc': rowConc,
247            'baseMeas': baseMeas
248        }
249        self._rowCounts = rowCounts
250
251        rowConc = self._idp['rowConc']
252        baseMeas = self._idp['baseMeas']
253        logDenom = logBeta(rowConc * baseMeas)
254        self._likelihoods = [
255            np.exp(logBeta(rowConc * baseMeas + row) - logDenom)
256            for row in rowCounts
257        ]
258
259        self._logScale = 0
260        self._weightedSims = []
261
262        self._savedSims = []
263        '''The state of `self._weightedSims` just prior to the last
264        successful call to `setEss`. Is reset by `addSims`, `clearSims`,
265        and `restoreSims`.
266        '''
267
268        self._allSims = []
269        '''The list of all simulations ever created by `addSims`.'''
270
271    @property
272    def colConc(self):
273        r'''`float`: The column concentration parameter. Should be
274        positive. Denoted by $\kappa$ in Section 9 of the Paper.
275        '''
276        return self._idp['colConc']
277
278    @property
279    def rowConc(self):
280        r'''`float`: The row concentration parameter. Should be
281        positive. Denoted by $\varepsilon$ in Section 9 of the Paper.

286    def baseMeas(self):
287        r'''`tuple`(`float`): The base probability vector. Should be a
288        sequence of positive numbers that sum to 1. Denoted by
289        $\boldsymbol{p} = (p_0, \ldots, p_{L - 1})$ in Section 9 of the
290        Paper, where $p_\ell = \varrho(\\{\ell\\})$.
291        '''
292        return tuple(self._idp['baseMeas'])
293

299        in Section 9 of the Paper. It is the number of observed
300        instances of `state` in the given row of data.
301        '''
302        return tuple(tuple(row) for row in self._rowCounts)
303

306        r'''`tuple`(`float`): The prior likelihoods. If `row` equals

308        $\varrho_N(A_m)$ in Section 9 of the Paper. It is the prior
309        probability of seeing the row counts for the given row.
310        '''
311        return tuple(self._likelihoods)

326    @property
327    def weightedSims(self):
328        r'''`tuple`(`WeightedSim`): A tuple of distinct (and therefore
329        independent) `WeightedSim` instances. `weightedSims`[k] is
330        denoted by $(t^k, \boldsymbol{\theta}_M^{*, k})$ in Section 9 of

333        '''
334        return tuple(self._weightedSims)

336    @property
337    def ess(self):
338        '''`float`: The effective sample size, approximated with the
339        sample variance. Denoted by $K_e''$ in Section 7.2 of the Paper.
340        '''
341        numSims = len(self._weightedSims)
342        if numSims == 0:
343            return 0
344        essPop = (
345            sum(sim.totalWeight      for sim in self._weightedSims) ** 2 /
346            sum(sim.totalWeight ** 2 for sim in self._weightedSims)
347        )
348        if essPop == numSims:
349            return numSims
350        return essPop * (numSims - 1) / (numSims - essPop / numSims)

383    def setEss(self, minEss, maxDel=None):
384        '''Tries to set the `ess` to the given value by deleting
385        simulations from `weightedSims`. Simulations are deleted in
386        order of their `WeightedSim.totalWeight` property, starting with
387        the largest.
388
389        **Arguments:**
390
391        * **minEss** (`float`): The minimum value of `ess` to achieve.
392            The method will stop deleting simulations when `ess` reaches
393            or exceeds this value.
394        * **maxDel** (`int`): If provided, the method will not delete
395            more than this number of simulations.
396
397        **Returns:**
398
399        * `int`: The number of simulations deleted.
400
401        **Raises:**
402
403        * **ValueError:** If
404            1. `minEss` does not exceed `ess`;
405            2. `minEss` is not less than the total number of simulations
406                in `weightedSims`;
407            3. `maxDel` is negative or zero; or
408            4. the method is unable to achieve the given value for
409                `ess`. In this case, all deletions are reversed and the
410                `weightedSims` property is left unaltered.
411
412        '''
413        savedSims = self._weightedSims.copy()
414        # Temporarily stored. Will be transferred to `self._savedSims`
415        # if the method finished without error.
416
417        if self.ess >= minEss:
418            raise ValueError(f'ESS is already {self.ess}')
419        if minEss >= len(self._weightedSims):
420            raise ValueError('Not enough simulations')
421        if maxDel is not None and maxDel <= 0:
422            raise ValueError('`maxDel` must be positive')
423
424        # make `minKeep`, the minimum number of simulations to keep
425        if maxDel is None:
426            minKeep = minEss
427        else:
428            minKeep = len(self._weightedSims) - maxDel
429
430        # try to reach `minEss` without deleting too many sims
431        sortedSims = sorted(
432            self._weightedSims,
433            key=lambda sim: sim.totalWeight,
434            reverse=True
435        )
436        for sim in sortedSims:
437            self._weightedSims.remove(sim)
438            if self.ess >= minEss or len(self._weightedSims) <= minKeep:
439                break
440
441        # check for success
442        if self.ess < minEss:
443            self._weightedSims = savedSims # undo deletions
444            raise ValueError(f'Cannot achieve an ESS of {minEss}')
445
446        self._savedSims = savedSims # on success, save for undo
447
448        return len(savedSims) - len(self._weightedSims)

466    def prior(self, mapping, numSamples=10000, bandwidth=None):
467        r'''Let the random vector $(X_0, X_1, \ldots, X_{L - 1})$ have a
468        Dirichlet distribution with parameters `colConc` * `baseMeas`.
469        This method returns the distribution of a random variable of the
470        form $Y = f(X_0, X_1, \ldots, X_{L - 1})$. The distribution of
471        $Y$ is a Borel probability measure on the real line. This method
472        therefore returns a `Measure` object.
473
474        If a function is passed to the `mapping` argument, then the cdf
475        and pdf of $Y$ are estimated by sampling and Gaussian kernel
476        density estimation (KDE). See `agentLaw` for details about
477        KDE.
478
479        This method is used in the implementation of (8.24), (8.26),
480        (9.6), and the final displayed equation in Section 9 of the
481        paper.
482
483        **Arguments:**
484
485        * **mapping** (`func` or `int`): The function to apply to the
486            random vector $(X_0, X_1, \ldots, X_{L - 1})$. Denoted by
487            $f$ in the above description. Passing this argument an
488            integer $\ell$ is equivalent to using the function $f(X_0,
489            X_1, \ldots, X_{L - 1}) = X_\ell$.
490        * **numSamples** (`int`): The number of samples to use when
491            estimating the cdf and pdf.
492        * **bandwidth** (`float`): The bandwidth to use in the KDE.
493
494        **Returns:**
495
496        * `Measure`: The distribution of $Y$.
497
498        '''
499        if not callable(mapping):
500            prior = beta(
501                self._idp['rowConc'] * self._idp['baseMeas'][mapping],
502                self._idp['rowConc'] * (1 - self._idp['baseMeas'][mapping])
503            )
504            return Measure(prior.cdf, prior.pdf, prior.mean())
505        rawSamples = dirichlet.rvs(
506            self._idp['rowConc'] * self._idp['baseMeas'],
507            numSamples
508        )
509        samples = [mapping(sample) for sample in rawSamples]
510        pdf = gaussian_kde(samples, bandwidth)
511        return Measure(
512            ecdf(samples).cdf.evaluate,
513            pdf,
514            np.average(samples),
515            pdf.factor
516        )

518    def law(self, mapping, newAgent=False, bandwidth=None):
519        r'''Let $\theta_{m\ell}$ be the unknown probability that agent
520        $m$ will perform action $\ell$, and let $\theta$ be the matrix
521        with entries $\theta_{m\ell}$. Let $\vartheta_\ell$ be the
522        unknown probability that a new, unobserved agent will perform
523        action $\ell$ This method returns the law (or distribution) of a
524        random variable of the form $\Theta = f(\vartheta_0, \ldots,
525        \vartheta_{L - 1})$, or of the form $\Theta = f(\theta)$. To get
526        the former, set `newAgent` to `True`. To get the latter, set
527        `newAgent` to `False`. In either case, set `mapping` to $f$.
528
529        The law of $\Theta$ is a Borel probability measure on the
530        real line. This method therefore returns a `Measure` object.
531        However, the law of $\Theta$ has a discrete component, so
532        it does not have a density. The `Measure.pdf` attribute of the
533        returned object is an approximate density, created with Gaussian
534        kernel density estimation (KDE). More specifically, it treats a
535        point mass $\delta_x$ as having a Gaussian density with mean $x$
536        and standard deviation $h$, where $h$ is the "bandwidth" of the
537        estimate.
538
539        This method implements either (8.24) or (8.10) in the Paper,
540        depending on whether `newAgent` is `True` or `False`,
541        respectively.
542
543        **Arguments:**
544
545        * **mapping** (`func`): The function $f$ in the above
546            description. This function should take a single `ndarray`
547            object as an argument. If `newAgent` is `True`, the array
548            should have the same shape as `baseMeas`. Otherwise, it
549            should have the same shape as `rowCounts`.
550        * **newAgent** (`bool`): Determines the form of $\Theta$ in the
551            above description.
552        * **bandwidth** (`float` or `None`): The bandwidth to use in the
553            Gaussian KDE. If `None`, the bandwidth will be calculated
554            using [Scott's Rule](https://doi.org/10.1002/9780470316849).
555            The calculated value can be retrieved by accessing the
556            `Measure.bandwidth` property of the returned object. This
557            argument is passed directly to the `gaussian_kde` function
558            in the `scipy.stats` module. See that documentation for
559            further details.
560
561        **Returns:**
562
563        * `Measure`: The law of $\Theta$.
564
565        **Raises:**
566
567        * **ValueError**: If the `weightedSims` property is empty.
568
569        '''
570        if not self._weightedSims:
571            raise ValueError('No weighted simulations found')
572
573        numRows = len(self._rowCounts)
574
575        if not newAgent:
576            # Implement (8.10).
577            samples = np.array([
578                mapping(weightedSim.rowDistSims)
579                for weightedSim in self._weightedSims
580            ])
581            weights = np.array([
582                weightedSim.totalWeight
583                for weightedSim in self._weightedSims
584            ])
585        else:
586            # Implement (8.24). In this case, the `sample` list is
587            # ordered as
588            # \[
589            #   \theta_1^{*, 1}, \ldots, \theta_M^{*, 1},
590            #   \theta_1^{*, 2}, \ldots, \theta_M^{*, 2},
591            #   \ldots,
592            #   \theta_1^{*, K}, \ldots, \theta_M^{*, K}
593            # \]
594            samples = np.array([
595                mapping(weightedSim.rowDistSims[m])
596                for weightedSim in self._weightedSims # outer loop
597                for m in range(numRows)  # inner loop
598            ])
599            weights = np.concatenate([
600                [weightedSim.totalWeight] * numRows
601                for weightedSim in self._weightedSims
602            ])
603
604        # Create the sample law
605        sampleCDF = ECDFDiscrete(samples, weights)
606        samplePDF = gaussian_kde(samples, bandwidth, weights)
607        sampleLaw = Measure(
608            sampleCDF,
609            samplePDF,
610            np.average(samples, weights=weights),
611            samplePDF.factor
612        )
613
614        # If implementing (8.10), return only the sample law. Otherwise,
615        # return a mixture as in (8.24).
616        if not newAgent:
617            return sampleLaw
618        colConc = self._idp['colConc']
619        return (
620            (colConc / (colConc + numRows)) * self.prior(mapping) +
621            (numRows / (colConc + numRows)) * sampleLaw
622        )

The probability density function of gamer is: $$ f(x) = \frac r {a^r \, \Gamma(a)} \, x^{-r - 1} \, \int_0^{ax} y^{a + r - 1} e^{-y} \, dy $$ for $x \ge 0$, $r > 0$, and $a > 0$. $\Gamma$ is the gamma function (scipy.special.gamma).

702
